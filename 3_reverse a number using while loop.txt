//3 reverse a number using while loop or palllindrome
#include<stdio.h>
#include<conio.h>
int main()
{
    int dividend,remainder=0,quotient=0,reverse=0,pallindrome;

    printf("input number ");
    scanf("%d",&dividend);
    pallindrome=dividend;
    printf("reverse number ");

    while(dividend>0){
        remainder=dividend%10;
        quotient=dividend/10;
        reverse=(reverse*10)+remainder;
        dividend=quotient;
    }
    printf("%d\n",reverse);
    if(pallindrome == reverse){
        printf("pallindrome");
    }else printf("not");
    return 0;
}


/*
1.check dividend>10
2.find remainder and print
3.find quotient and update dividend=quotient
4.last printf for print last dividend
5. formula divisor|dividend|quotient
                -------------
                  remainder
*/
